async loadWeb3() {
    let ethereum

    if (window.ethereum) {
    ethereum = window.ethereum
    const web3 = new Web3(ethereum)
    try {
        await ethereum.enable().then(function(){
        User allows account access to DApp from browser
        console.log("Address",web3.eth.getAccounts()[0])

        console.log("User Connected!")
        });
    } catch (e) {
        User denies account access to DApp from browser
        console.log("ERROR",e)
    }
    } else {
    alert("You need to install Metamask to use this application")
    }
}
